# 3rd Solution:
```
class A {
  public:
    int x;
    A(int x) : x(x) {}
    A(A* obj1) : x(obj1->x){}
};

class B : public A {
  public:
    int y;
    B(int x, int y) : A(x), y(y) {}
    B(B* obj1) : A(obj1), y(obj1->y){}
};


int main()
{
    A* obj_a = new A(5);
    B* obj_b1 = new B(6, 7);
    
    B* obj_b2 = new B(obj_b1);
    
    cout<<obj_b2->x<<" "<<obj_b2->y;
    return 0;
}
```
